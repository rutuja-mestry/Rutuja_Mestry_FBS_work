FROM (SELECT COUNT(*) cnt FROM Issue GROUP BY Member_Id) t);

FROM Member m
LEFT JOIN Issue i ON m.Member_Id=i.Member_Id
WHERE m.Membership_type='Lifetime'

GROUP BY m.Member_Id
HAVING COUNT(i.Book_No)=1;
